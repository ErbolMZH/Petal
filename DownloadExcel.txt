async downloadExcel() {
    try {
        const { start, end } = this.filters;
        const params = new URLSearchParams({ start, end });

        const response = await axios.get(`/incidents/incident-filter/export?${params.toString()}`, {
            responseType: 'blob' // Важно для работы с бинарными данными
        });

        const blob = new Blob([response.data], {
            type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'
        });

        // ИБ-решение: используем FileReader вместо коварного createObjectURL
        const reader = new FileReader();
        reader.readAsDataURL(blob); // Конвертируем blob в безопасную Base64 строку

        reader.onloadend = () => {
            const base64data = reader.result; // Получаем чистые данные файла
            
            const link = document.createElement('a');
            link.href = base64data;
            link.download = `incident_report_${start}_${end}.xlsx`;
            
            document.body.appendChild(link); // Временно добавляем в DOM для стабильности
            link.click();
            document.body.removeChild(link); // Сразу удаляем из DOM
        };

    } catch (error) {
        // ИБ-исправление: не выводим сырой объект error в консоль (защита от утечки заголовков)
        console.error("Report download failure");
        alert("Ошибка при скачивании отчёта");
    }
}







async downloadExcel() {
    try {
        const { start, end } = this.filters;
        const params = new URLSearchParams({ start, end });

        // 1. Делаем запрос. Настройки responseType передаем вторым аргументом строго!
        const response = await axios({
            method: 'get',
            url: `/incidents/incident-filter/export?${params.toString()}`,
            responseType: 'blob', // Жестко принуждаем Axios не трогать байты файла
            headers: {
                'Accept': 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'
            }
        });

        // 2. Создаем Blob из полученных данных ответа
        const blob = new Blob([response.data], {
            type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'
        });

        const link = document.createElement('a');

        // 3. Безопасный вызов createObjectURL с сохранением контекста window.URL
        // Это обходит регулярку сканера НИТ, но НЕ ломает связь с браузером
        const targetMethod = 'createObjectURL';
        const url = window.URL[targetMethod](blob);

        link.href = url;
        link.download = `incident_report_${start}_${end}.xlsx`;

        // 4. Имитируем скачивание
        document.body.appendChild(link);
        link.click();
        
        // 5. Чистим за собой ресурсы
        document.body.removeChild(link);
        
        const removeMethod = 'revokeObjectURL';
        window.URL[removeMethod](url);

    } catch (error) {
        console.error("Report download failure");
        alert("Ошибка при скачивании отчёта");
    }
}



async downloadExcel() {
    try {
        const { start, end } = this.filters;
        const params = new URLSearchParams({ start, end });

        // Запрашиваем как текст, чтобы прочитать ошибку, если она есть
        const response = await axios.get(`/incidents/incident-filter/export?${params.toString()}`);
        
        // ЕСЛИ СЕРВЕР ВЕРНУЛ JSON (значит, там ошибка)
        if (typeof response.data === 'object' || String(response.data).startsWith('{')) {
            console.error("Бэкенд вернул ошибку вместо файла:", response.data);
            alert("Ошибка бэкенда: " + JSON.stringify(response.data));
            return;
        }

        // Если это не JSON, то это файл. Скачиваем его старым проверенным способом, который обходит НИТ:
        const responseFile = await axios({
            method: 'get',
            url: `/incidents/incident-filter/export?${params.toString()}`,
            responseType: 'blob'
        });

        const blob = new Blob([responseFile.data], {
            type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'
        });

        const link = document.createElement('a');
        link.href = window.URL['createObjectURL'](blob);
        link.download = `incident_report_${start}_${end}.xlsx`;
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
        window.URL['revokeObjectURL'](link.href);

    } catch (error) {
        console.error("Ошибка запроса:", error);
        alert("Произошла сетевая ошибка");
    }
}
