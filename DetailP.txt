async fetchPersonInfo() {
    this.isLoading = true;
    this.hasError = false;
    try {
        const resp = await axios.get(`/incidents/person/${this.person_id}`);
        let personData = resp.data.data;

        // Вставьте сюда точный пароль из вашей SQL функции f_get_key()
        const secretKey = "СЮДА_ВСТАВЬТЕ_ВАШ_КЛЮЧ"; 

        // Массив гарантированно лежит в personData.ibans
        if (personData.ibans && personData.ibans.length > 0) {
            console.log(`Найдено ${personData.ibans.length} зашифрованных IBAN. Начинаем дешифрацию...`);
            personData.ibans = await this.decryptIbans(personData.ibans, secretKey);
        }

        this.person = personData;

    } catch (error) {
        this.hasError = true;
        this.errorMessage = error.response ? error.response.data : error;
    } finally {
        this.isLoading = false;
    }
},

async decryptIbans(ibansList, secretKey) {
    const decryptedList = [];

    for (let item of ibansList) {
        try {
            if (item.iban) {
                let rawIban = item.iban.trim();
                console.log("Шифр-текст из базы:", rawIban);

                let bytes;

                // 1. Декодируем данные в зависимости от формата Postgres
                if (rawIban.startsWith('\\x')) {
                    // Удаляем префикс '\x'
                    let hexString = rawIban.slice(2);
                    
                    // Проверяем, валидный ли это HEX (только 0-9, a-f)
                    const isHex = /^[0-9a-fA-F]+$/.test(hexString);

                    if (isHex) {
                        // Вариант А: Чистый HEX формат
                        const len = hexString.length;
                        bytes = new Uint8Array(len / 2);
                        for (let i = 0, j = 0; i < len; i += 2, j++) {
                            bytes[j] = parseInt(hexString.substr(i, 2), 16);
                        }
                    } else {
                        // Вариант Б: Если Postgres вернул строку в формате 'escape' 
                        // (когда бинарные данные переводятся в ASCII-подобную строку)
                        bytes = new TextEncoder().encode(rawIban);
                    }
                } else {
                    // Вариант В: Обычная строка (если Go уже превратил байты в строку текста)
                    bytes = new TextEncoder().encode(rawIban);
                }

                // 2. Читаем OpenPGP сообщение из полученных байтов
                const message = await window.openpgp.readMessage({
                    binaryMessage: bytes 
                });

                // 3. Расшифровываем с помощью секретного ключа
                const { data: decryptedText } = await window.openpgp.decrypt({
                    message: message,
                    passwords: [secretKey],
                    format: 'text'
                });

                console.log("УРА! Результат расшифровки:", decryptedText);
                item.iban = decryptedText; // Подменяем зашифрованную строку на чистый IBAN
            }
        } catch (decryptError) {
            console.error("Не удалось расшифровать этот IBAN. Ошибка:", decryptError);
            item.iban = "Ошибка расшифровки"; 
        }
        decryptedList.push(item);
    }

    return decryptedList;
}
