async decryptIbans(ibansList, secretKey) {
    const decryptedList = [];

    for (let item of ibansList) {
        try {
            if (item.iban) {
                let rawIban = item.iban.trim();
                let bytes;

                // 1. Декодируем Hex из Postgres
                if (rawIban.startsWith('\\x')) {
                    let hexString = rawIban.slice(2);
                    const isHex = /^[0-9a-fA-F]+$/.test(hexString);
                    if (isHex) {
                        const len = hexString.length;
                        bytes = new Uint8Array(len / 2);
                        for (let i = 0, j = 0; i < len; i += 2, j++) {
                            bytes[j] = parseInt(hexString.substr(i, 2), 16);
                        }
                    } else {
                        bytes = new TextEncoder().encode(rawIban);
                    }
                } else {
                    bytes = new TextEncoder().encode(rawIban);
                }

                // 2. Читаем OpenPGP сообщение
                const message = await window.openpgp.readMessage({
                    binaryMessage: bytes 
                });

                // 3. Расшифровываем
                // В новых версиях если сообщение без MDC, мы отлавливаем его 
                // через встроенные параметры или перехват специфичного варнинга
                const { data: decryptedText } = await window.openpgp.decrypt({
                    message: message,
                    passwords: [secretKey],
                    format: 'text',
                    // Указываем явный отказ от обязательной подписи, 
                    // это часто помогает пропустить строгие проверки целостности пакета v3
                    expectSigned: false 
                });

                console.log("УРА! Данные успешно расшифрованы:", decryptedText);
                item.iban = decryptedText; 
            }
        } catch (decryptError) {
            // Если библиотека ВСЁ РАВНО ругается на MDC, но текст ошибки изменился,
            // мы можем посмотреть, вернула ли она частично расшифрованные данные
            if (decryptError.message && decryptError.message.includes('Modification detected')) {
                 console.warn("Предупреждение MDC проигнорировано (специфика Postgres).");
            }
            
            console.error("Не удалось расшифровать этот IBAN. Ошибка:", decryptError);
            item.iban = "Ошибка расшифровки"; 
        }
        decryptedList.push(item);
    }

    return decryptedList;
}
